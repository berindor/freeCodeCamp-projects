function destroyer(arr) {
  const newArr = [...arr];
  for (let elem = arr.length - 1; elem >= 0; elem--) {
    for (let arg = 1; arg < arguments.length; arg++) {
      if (arr[elem] === arguments[arg]) {
        newArr.splice(elem, 1);
      }
    }
  }
  console.log(newArr);
  return newArr;
}
